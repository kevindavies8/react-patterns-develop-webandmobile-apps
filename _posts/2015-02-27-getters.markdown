Getters **SHOULD** be used when accessing computed properties.

#### bad
{% highlight js %}
fullName() {
  return `${this.props.firstName} {this.props.lastName}`;
}

render() {
  return <h3>{this.fullName()}</h3>;
}

{% endhighlight %}

#### good
{% highlight js %}
get fullName() {
  return `${this.props.firstName} {this.props.lastName}`;
}

render() {
  return <h3>{this.fullName}</h3>;
}
{% endhighlight %}
